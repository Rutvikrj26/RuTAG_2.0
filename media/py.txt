from django.db import models

# Create your models here.
class projects(models.Model):
    Projecttitle = models.CharField(max_length=100)
    Projectphoto = models.CharField(max_length=1000)
    Projectvideo  = models.CharField(max_length=1000)
    Projectwriteup = models.CharField(max_length=1000)

    def __str__(self):
        return(self.Projecttitle)

class Professors(models.Model):
    projects = models.ForeignKey(projects,on_delete=models.CASCADE)
    professors =models.CharField(max_length=25)

class Students(models.Model):
    projects = models.ForeignKey(projects, on_delete=models.CASCADE)
    students = models.CharField(max_length=25)
